```typescript
import { profile, type ProfileData } from '../data/profile';

/**
 * Mount the profile section into #profile and set up the Bing background.
 * Handles: avatar onerror fallback, Bing image preloading with load/error states,
 * and bio text rendering with emoji support.
 */
export function renderProfile(data: ProfileData): void {
  const profileEl = document.getElementById('profile');
  if (!profileEl) return;

  // --- Create and inject the page background element ---
  const bgEl = document.createElement('div');
  bgEl.className = 'page-bg';
  bgEl.setAttribute('aria-hidden', 'true');
  document.body.prepend(bgEl);

  // Preload the Bing background image to detect load/error
  const preloader = new Image();
  preloader.src = data.backgroundUrl;
  preloader.onload = () => {
    bgEl.classList.add('loaded');
  };
  preloader.onerror = () => {
    // Keep the gradient placeholder (already visible as .page-bg background).
    // Do NOT add 'loaded' class -- this keeps ::before opacity at 0 and the
    // gradient placeholder remains visible. Per D-19: solid background-color
    // fallback if Bing image fails.
  };

  // --- Avatar ---
  const avatar = document.createElement('img');
  avatar.src = data.avatarUrl;
  avatar.alt = "Sy's avatar";
  avatar.className = 'profile-avatar';
  avatar.width = 96;
  avatar.height = 96;
  avatar.loading = 'lazy';
  avatar.decoding = 'async';

  // Fallback on error (PROF-01, Pitfall 4)
  avatar.onerror = () => {
    const fallback = document.createElement('div');
    fallback.className = 'profile-avatar-fallback';
    fallback.textContent = 'Sy';
    fallback.setAttribute('aria-label', "Sy's avatar (fallback)");
    avatar.replaceWith(fallback);
  };

  // --- Username ---
  const username = document.createElement('h1');
  username.className = 'profile-username';
  username.textContent = data.username; // "Sy"

  // --- Bio ---
  const bio = document.createElement('p');
  bio.className = 'profile-bio';
  bio.textContent = data.bio;

  // --- Assemble ---
  profileEl.appendChild(avatar);
  profileEl.appendChild(username);
  profileEl.appendChild(bio);
}
```

Key design decisions:
- The `.page-bg` div is prepended to `<body>`, not appended to `#profile`. This makes it a fixed full-page background layer (z-index: 0 in CSS). The profile, timeline, and footer sit on top (z-index: 10).
- The `Image()` preloader detects whether the Bing URL resolves. On success, it adds the `loaded` class, triggering CSS opacity transition on `::before` from 0 to 1 (0.6s ease-in-out). On error, the gradient placeholder persists.
- The avatar `onerror` handler creates a replacement `<div>` showing initials "Sy" with accent background color. The `replaceWith()` call swaps the broken `<img>` for the fallback div in-place.
- Explicit `width`/`height` on avatar prevents Cumulative Layout Shift (CLS) when the image loads (PITFALLS.md technical debt avoidance).
- `loading="lazy"` and `decoding="async"` on the avatar to avoid blocking the main thread (PITFALLS.md avoidance).
- `bio.textContent` (not innerHTML) for security -- the bio contains user-provided text but no HTML. Using textContent prevents any accidental HTML injection.
- The function signature `renderProfile(data: ProfileData): void` is the contract for Plan 04's main.ts to call.

Do NOT: call any API, use any framework, add React/Vue/Svelte, or manually write CSS inline styles (all styling is in styles.css).
